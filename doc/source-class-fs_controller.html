 390:    /**
 391:     * Una IP será baneada si falla más de 5 intentos de login en menos de 10 minutos
 392:     * @param type $ips es un array de IP;intentos;hora
 393:     * @return boolean
 394:     */
 395:    private function ip_baneada(&$ips)
 396:    {
 397:       $baneada = FALSE;
 398:       
 399:       if( file_exists('tmp/'.FS_TMP_NAME.'ip.log') )
 400:       {
 401:          $file = fopen('tmp/'.FS_TMP_NAME.'ip.log', 'r');
 402:          if($file)
 403:          {
 404:             /// leemos las líneas
 405:             while( !feof($file) )
 406:             {
 407:                $linea = explode(';', trim(fgets($file)));
 408:                
 409:                if( intval($linea[2]) > time() )
 410:                {
 411:                   if($linea[0] == $_SERVER['REMOTE_ADDR'] AND intval($linea[1]) > 5)
 412:                   {
 413:                      $baneada = TRUE;
 414:                      
 415:                      if( intval($linea[1]) == 6 )
 416:                      {
 417:                         $fslog = new fs_log();
 418:                         
 419:                         if( isset($_POST['user']) )
 420:                         {
 421:                            $fslog->usuario = $_POST['user'];
 422:                         }
 423:                         
 424:                         $fslog->tipo = 'login';
 425:                         $fslog->detalle = 'IP baneada';
 426:                         $fslog->ip = $_SERVER['REMOTE_ADDR'];
 427:                         $fslog->alerta = TRUE;
 428:                         $fslog->save();
 429:                      }
 430:                   }
 431:                   
 432:                   $ips[] = $linea;
 433:                }
 434:             }
 435:             
 436:             fclose($file);
 437:          }
 438:       }
 439:       
 440:       return $baneada;
 441:    }

 443:    /**
 444:     * Baneamos las IPs que fallan más de 5 intentos de login en 10 minutos
 445:     * @param type $ips es un array de IP;intentos;hora
 446:     */
 447:    private function banear_ip(&$ips)
 448:    {
 449:       $file = fopen('tmp/'.FS_TMP_NAME.'ip.log', 'w');
 450:       if($file)
 451:       {
 452:          $encontrada = FALSE;
 453:          
 454:          foreach($ips as $ip)
 455:          {
 456:             if($ip[0] == $_SERVER['REMOTE_ADDR'])
 457:             {
 458:                fwrite( $file, $ip[0].';'.( 1+intval($ip[1]) ).';'.( time()+600 ) );
 459:                $encontrada = TRUE;
 460:             }
 461:             else
 462:                fwrite( $file, join(';', $ip) );
 463:          }
 464:          
 465:          if(!$encontrada)
 466:             fwrite( $file, $_SERVER['REMOTE_ADDR'].';1;'.( time()+600 ) );
 467:          
 468:          fclose($file);
 469:       }
 470:    }

 477:    {

 480:       if( $this->ip_baneada($ips) )

 482:          $this->banear_ip($ips);

 484:       }

 486:       {
 487:          if( FS_DEMO ) /// en el modo demo nos olvidamos de la contraseña
 488:          {
 489:             $user = $this->user->get($_POST['user']);
 490:             if( !$user )
 491:             {
 492:                $user = new fs_user();
 493:                $user->nick = $_POST['user'];
 494:                $user->set_password('demo');
 495:                $user->admin = TRUE;
 496:                
 497:                /// creamos un agente para asociarlo
 498:                $agente = new agente();
 499:                $agente->codagente = $agente->get_new_codigo();
 500:                $agente->nombre = $_POST['user'];
 501:                $agente->apellidos = 'Demo';
 502:                if( $agente->save() )
 503:                   $user->codagente = $agente->codagente;
 504:             }
 505:             
 506:             $user->new_logkey();
 507:             if( $user->save() )
 508:             {
 509:                setcookie('user', $user->nick, time()+FS_COOKIES_EXPIRE);
 510:                setcookie('logkey', $user->log_key, time()+FS_COOKIES_EXPIRE);
 511:                $this->user = $user;
 512:                $this->load_menu();
 513:             }
 514:          }
 515:          else
 516:          {
 517:             $user = $this->user->get($_POST['user']);
 518:             $password = strtolower($_POST['password']);
 519:             if($user)
 520:             {
 521:                if( $user->password == sha1($password) )
